        function getRandomArbitrary(min, max) {
            min = Math.ceil(min);
            max = Math.floor(max);
            return Math.floor(Math.random() * (max - min)) + min;
        }

                Possible prize on current attempt:  ${startSumm}$`);
                flag = false;
                if(usernumber === randomNumber){
                    flag = true;
                    switch (i) {

                            break;

                            break;

                    console.log(`${totalPrize}$ -true`);

                }
            }
            if (flag){
